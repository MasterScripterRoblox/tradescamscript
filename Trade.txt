-- Function to generate a random string of given length
local function getRandomString(length)
    local chars = "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789"
    local result = ""
    for i = 1, length do
        local randomIndex = math.random(1, #chars)
        result = result .. chars:sub(randomIndex, randomIndex)
    end
    return result
end

-- Copy 3 random strings to the clipboard (one by one)
for i = 1, 50 do
    local randomString = getRandomString(10) -- 10-character random string
    setclipboard(randomString) -- Copy to clipboard
    print("Copied to clipboard: " .. randomString)
    wait(0.5) -- small delay to avoid issues
end
